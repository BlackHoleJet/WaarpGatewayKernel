107 	/**
108 	 * Called at the beginning of every request to get the current HttpBusinessFactory to use.
109 	 * 
110 	 * @param remoteAddress
111 	 *            the remote socket address in use
112 	 * @return AbstractHttpBusinessRequest to use during the request
113 	 */
114 	public AbstractHttpBusinessRequest newRequest(SocketAddress remoteAddress) {
115 		LinkedHashMap<String, AbstractHttpField> linkedHashMap = new LinkedHashMap<String, AbstractHttpField>(
116 				this.fields.size());
117 		for (AbstractHttpField field : this.fields.values()) {
118 			AbstractHttpField newfield = field.clone();
119 			if (pagerole != PageRole.MENU) {
120 				newfield.fieldvalue = null;
121 			}
122 			linkedHashMap.put(field.fieldname, newfield);
123 		}
124 		return this.httpBusinessFactory.getNewHttpBusinessRequest(remoteAddress, linkedHashMap,
125 				this);
126 	}

128 	public String getPageValue(String value) {
129 		if (this.fileform != null && value != null) {
130 			try {
131 				return WaarpStringUtils.readFileException(fileform + value);
132 			} catch (InvalidArgumentException e) {
133 			} catch (FileTransferException e) {
134 			}
135 		}
136 		return value;
137 	}

139 	/**
140 	 * 
141 	 * @param reference
142 	 * @return the Html results for all pages except Get (result of a download)
143 	 * @throws HttpIncorrectRequestException
144 	 */
145 	public String getHtmlPage(AbstractHttpBusinessRequest reference)
146 			throws HttpIncorrectRequestException {
147 		if (this.pagerole == PageRole.HTML) {
148 			// No handling of variable management, use MENU instead
149 			String value = reference.getHeader();
150 			logger.debug("Debug: " + (value != null));
151 			if (value == null || value.length() == 0) {
152 				value = getPageValue(this.header);
153 			}
154 			StringBuilder builder = null;
155 			if (value == null) {
156 				builder = new StringBuilder();
157 			} else {
158 				builder = new StringBuilder(value);
159 			}
160 			value = reference.getBeginForm();
161 			if (value == null || value.length() == 0) {
162 				value = getPageValue(this.beginform);
163 			}
164 			if (value != null) {
165 				builder.append(value);
166 			}
167 			value = reference.getEndForm();
168 			if (value == null || value.length() == 0) {
169 				value = getPageValue(this.endform);
170 			}
171 			if (value != null) {
172 				builder.append(value);
173 			}
174 			value = reference.getFooter();
175 			if (value == null || value.length() == 0) {
176 				value = getPageValue(this.footer);
177 			}
178 			if (value != null) {
179 				builder.append(value);
180 			}
181 			return builder.toString();
182 		}
183 		boolean isForm = reference.isForm();
184 		String value = reference.getHeader();
185 		if (value == null || value.length() == 0) {
186 			value = getPageValue(this.header);
187 		}
188 		StringBuilder builder = null;
189 		if (value == null) {
190 			builder = new StringBuilder();
191 		} else {
192 			builder = new StringBuilder(value);
193 		}
194 		LinkedHashMap<String, AbstractHttpField> requestFields = reference
195 				.getLinkedHashMapHttpFields();
196 		if (!isForm) {
197 			value = reference.getBeginForm();
198 			if (value == null || value.length() == 0) {
199 				value = getPageValue(this.beginform);
200 			}
201 			if (value != null) {
202 				builder.append(value);
203 			} else {
204 				builder.append("<BR><TABLE border=1><TR>");
205 				for (AbstractHttpField field : requestFields.values()) {
206 					if (field.fieldvisibility) {
207 						builder.append("<TH>");
208 						builder.append(field.fieldinfo);
209 						builder.append("</TH>");
210 					}
211 				}
212 				builder.append("<TR>");
213 			}
214 		} else {
215 			value = reference.getBeginForm();
216 			if (value == null || value.length() == 0) {
217 				value = getPageValue(this.beginform);
218 			}
219 			if (value != null) {
220 				builder.append(value);
221 			}
222 		}
223 		boolean first = true;
224 		for (AbstractHttpField field : requestFields.values()) {
225 			if (field.fieldvisibility) {
226 				// to prevent that last will have a next field form
227 				if (first) {
228 					first = false;
229 				} else {
230 					value = reference.getNextFieldInForm();
231 					if (value == null || value.length() == 0) {
232 						value = getPageValue(this.nextinform);
233 					}
234 					if (value != null) {
235 						builder.append(value);
236 					}
237 				}
238 				value = reference.getFieldForm(field);
239 				if (value == null || value.length() == 0) {
240 					if (isForm) {
241 						value = field.getHtmlFormField(this);
242 					} else {
243 						value = field.getHtmlTabField(this);
244 					}
245 				}
246 				if (value != null) {
247 					builder.append(value);
248 				}
249 			}
250 		}
251 		if (!isForm) {
252 			value = reference.getEndForm();
253 			if (value == null || value.length() == 0) {
254 				value = getPageValue(this.endform);
255 			}
256 			if (value != null) {
257 				builder.append(value);
258 			} else {
259 				builder.append("</TABLE><BR>");
260 			}
261 		} else {
262 			value = reference.getEndForm();
263 			if (value == null || value.length() == 0) {
264 				value = getPageValue(this.endform);
265 			}
266 			if (value != null) {
267 				builder.append(value);
268 			}
269 		}
270 		value = reference.getFooter();
271 		if (value == null || value.length() == 0) {
272 			value = getPageValue(this.footer);
273 		}
274 		if (value != null) {
275 			builder.append(value);
276 		}
277 		return builder.toString();
278 	}

280 	/**
281 	 * Set the value to the field according to fieldname.
282 	 * 
283 	 * If the field is not registered, the field is ignored.
284 	 * 
285 	 * @param reference
286 	 * @param fieldname
287 	 * @param value
288 	 * @param position
289 	 * @throws HttpIncorrectRequestException
290 	 */
291 	public void setValue(AbstractHttpBusinessRequest reference, String fieldname, String value,
292 			FieldPosition position)
293 			throws HttpIncorrectRequestException {
294 		LinkedHashMap<String, AbstractHttpField> requestFields = reference
295 				.getLinkedHashMapHttpFields();
296 		AbstractHttpField field = requestFields.get(fieldname);
297 		if (field != null) {
298 			if (field.fieldposition == FieldPosition.ANY || field.fieldposition == position) {
299 				field.setStringValue(value);
300 				if (field.fieldtovalidate) {
301 					if (!reference.isFieldValid(field)) {
302 						throw new HttpIncorrectRequestException("Field unvalid: " + fieldname);
303 					}
304 				}
305 			} else {
306 				throw new HttpIncorrectRequestException("Invalid position: " + position +
307 						" while field is supposed to be in " + field.fieldposition);
308 			}
309 		}
310 	}

337 	/**
338 	 * 
339 	 * @param reference
340 	 * @return True if the request is fully valid
341 	 */
342 	public boolean isRequestValid(AbstractHttpBusinessRequest reference) {
343 		LinkedHashMap<String, AbstractHttpField> requestFields = reference
344 				.getLinkedHashMapHttpFields();
345 		for (AbstractHttpField field : requestFields.values()) {
346 			if (field.fieldmandatory && !field.present) {
347 				logger.warn("Request invalid since the following field is absent: "
348 						+ field.fieldname);
349 				return false;
350 			}
351 		}
352 		return reference.isRequestValid();
353 	}
