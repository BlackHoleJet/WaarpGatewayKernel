100     public static enum METHOD {
101     	/**
102     	 * REST: Standard GET item
103     	 * 
104     	 * The GET method means retrieve whatever information (in the form of an entity) 
105     	 * is identified by the Request-URI. If the Request-URI refers to a 
106     	 * data-producing process, it is the produced data which shall be returned as 
107     	 * the entity in the response and not the source text of the process, unless 
108     	 * that text happens to be the output of the process.
109     	 */
110     	GET(HttpMethod.GET),
111     	/**
112     	 * REST: Update existing item
113     	 * 
114     	 * The PUT method requests that the enclosed entity be stored under the 
115     	 * supplied Request-URI.
116     	 */
117     	PUT(HttpMethod.PUT),
118     	/**
119     	 * REST: Create a new item
120     	 * 
121     	 * The POST method is used to request that the origin server accept the 
122     	 * entity enclosed in the request as a new subordinate of the resource 
123     	 * identified by the Request-URI in the Request-Line.
124     	 */
125     	POST(HttpMethod.POST), 
126     	/**
127     	 * REST: Delete existing item
128     	 * 
129     	 * The DELETE method requests that the origin server delete the resource 
130     	 * identified by the Request-URI.
131     	 */
132     	DELETE(HttpMethod.DELETE), 
133     	/**
134     	 * REST: what options are supported for the URI
135     	 * 
136     	 * The OPTIONS method represents a request for information about 
137     	 * the communication options available on the request/response chain 
138     	 * identified by the Request-URI. This method allows the client to 
139     	 * determine the options and/or requirements associated with a resource, 
140     	 * or the capabilities of a server, without implying a resource action 
141     	 * or initiating a resource retrieval.
142     	 */
143     	OPTIONS(HttpMethod.OPTIONS), 
144     	/**
145     	 * REST: as GET but no BODY (existence ? metadata ?)
146     	 * 
147     	 * The HEAD method is identical to GET except that the server MUST NOT 
148     	 * return a message-body in the response.
149     	 */
150     	HEAD(HttpMethod.HEAD),
151     	/**
152     	 * REST: should not be used, use POST instead
153     	 * 
154     	 * The PATCH method requests that a set of changes described in the 
155     	 * request entity be applied to the resource identified by the Request-URI.
156     	 */
157     	PATCH(HttpMethod.PATCH),
158     	/**
159     	 * REST: unknown usage
160     	 * 
161     	 * The TRACE method is used to invoke a remote, application-layer 
162     	 * loop-back of the request message.
163     	 */
164     	TRACE(HttpMethod.TRACE),
165     	/**
166     	 * REST: unknown
167     	 * 
168     	 * This specification reserves the method name CONNECT for use with 
169     	 * a proxy that can dynamically switch to being a tunnel
170     	 */
171     	CONNECT(HttpMethod.CONNECT);
172     	
173     	public final HttpMethod method;
174     	private METHOD(HttpMethod method) {
175     		this.method = method;
176     	}
177     }

274 	 * 
275 	 * Override if needed
276 	 */
277 	protected void clean() {
278 		if (arguments != null) {
279 			arguments.clean();
280 			arguments = null;
281 		}
282 		if (response != null) {
283 			response.clean();
284 			response = null;
285 		}
286 		if (decoder != null) {
287 			decoder.cleanFiles();
288 			decoder = null;
289 		}
290 		handler = null;
291 		cumulativeBody = null;
292 		jsonObject = null;
293 		if (dbSession != null) {
294 			dbSession.enUseConnectionNoDisconnect();
295 			dbSession = null;
296 		}

333 	/**
334 	 * Method to set Cookies in httpResponse from response ObjectNode
335 	 * 
336 	 * @param httpResponse
337 	 */
338 	protected void setCookies(HttpResponse httpResponse) {
339 		if (response == null) {
340 			return;
341 		}
342 		ObjectNode cookieON = response.getCookieArgs();
343 		if (! cookieON.isMissingNode()) {
344 			Iterator<Entry<String, JsonNode>> iter = cookieON.fields();
345 			while (iter.hasNext()) {
346 				CookieEncoder cookieEncoder = new CookieEncoder(true);
347 				Entry<String, JsonNode> entry = iter.next();
348 				cookieEncoder.addCookie(entry.getKey(), entry.getValue().asText());
349 				httpResponse.headers().add(HttpHeaders.Names.SET_COOKIE, cookieEncoder.encode());
350 			}
351 		}
352 	}

354 	/**
355 	 * Could be overwritten if necessary
356 	 * 
357 	 * @return RestMethodHandler associated with the current context
358 	 * @throws HttpIncorrectRequestException 
359 	 * @throws HttpMethodNotAllowedRequestException 
360 	 * @throws HttpForbiddenRequestException 
361 	 */
362 	protected RestMethodHandler getHandler() throws HttpMethodNotAllowedRequestException, HttpForbiddenRequestException {
363 		METHOD method = arguments.getMethod();
364 		String uri = arguments.getBaseUri();
365 		boolean restFound = false;
366 		RestMethodHandler handler = restHashMap.get(uri);
367 		if (handler != null) {
368 			handler.checkHandlerSessionCorrectness(this, arguments, response);
369 			if (handler.isMethodIncluded(method)) {
370 				restFound = true;
371 			}
372 		}
373 		if (handler == null && method == METHOD.OPTIONS) {
374 			handler = rootHandler;
375 			// use Options default handler
376 			restFound = true;
377 		}
378 		if (! restFound){
379 			throw new HttpMethodNotAllowedRequestException("No Method found for that URI: "+uri);
380 		}
381 		return handler;
382 	}

384 	@Override
385 	public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) {
386         logger.debug("Msg Received");
387 		Channel channel = ctx.getChannel();
388 		try {
389 			if (!readingChunks) {
390 				initialize();
391 				this.request = (HttpRequest) e.getMessage();
392 				arguments.setRequest(request);
393 				arguments.setHeaderArgs(request.headers().entries());
394 				arguments.setCookieArgs(request.headers().get(HttpHeaders.Names.COOKIE));
395 				logger.debug("DEBUG: {}", arguments);
396 				checkConnection(channel);
397 				handler = getHandler();
398 				if (arguments.getMethod() == METHOD.OPTIONS) {
399 					response.setFromArgument(arguments);
400 					handler.optionsCommand(this, arguments, response);
401 					finalizeSend(channel);
402 					return;
403 				}
404 				if (request.isChunked()) {
405 					// no body yet
406 					readingChunks = true;
407 					if (! handler.isBodyJsonDecoded()) {
408 						createDecoder();
409 					}
410 					return;
411 				} else {
412 					if (handler.isBodyJsonDecoded()) {
413 						ChannelBuffer buffer = request.getContent();
414 						jsonObject = getBodyJsonArgs(buffer);
415 					} else {
416 						// decoder for 1 chunk
417 						createDecoder();
418 						// Not chunk version
419 						readAllHttpData();
420 					}
421 					response.setFromArgument(arguments);
422 					handler.endParsingRequest(this, arguments, response, jsonObject);
423 					finalizeSend(channel);
424 					return;
425 				}
426 			} else {
427 				// New chunk is received
428 				bodyChunk(e);
429 			}
430 		} catch (HttpIncorrectRequestException e1) {
431 			// real error => 400
432 			if (handler != null) {
433 				status = handler.handleException(this, arguments, response, jsonObject, e1);
434 			}
435 			if (status == HttpResponseStatus.OK) {
436 				status = HttpResponseStatus.BAD_REQUEST;
437 			}
438 			logger.warn("Error", e1);
439 			if (response.getDetail().isEmpty()) {
440 				response.setDetail(e1.getMessage());
441 			}
442 			if (handler != null) {
443 				finalizeSend(channel);
444 			} else {
445 				forceClosing(e.getChannel());
446 			}
447 		} catch (HttpMethodNotAllowedRequestException e1) {
448 			if (handler != null) {
449 				status = handler.handleException(this, arguments, response, jsonObject, e1);
450 			}
451 			if (status == HttpResponseStatus.OK) {
452 				status = HttpResponseStatus.METHOD_NOT_ALLOWED;
453 			}
454 			logger.warn("Error", e1);
455 			if (response.getDetail().isEmpty()) {
456 				response.setDetail(e1.getMessage());
457 			}
458 			if (handler != null) {
459 				finalizeSend(channel);
460 			} else {
461 				forceClosing(e.getChannel());
462 			}
463 		} catch (HttpForbiddenRequestException e1) {
464 			if (handler != null) {
465 				status = handler.handleException(this, arguments, response, jsonObject, e1);
466 			}
467 			if (status == HttpResponseStatus.OK) {
468 				status = HttpResponseStatus.FORBIDDEN;
469 			}
470 			logger.warn("Error", e1);
471 			if (response.getDetail().isEmpty()) {
472 				response.setDetail(e1.getMessage());
473 			}
474 			if (handler != null) {
475 				finalizeSend(channel);
476 			} else {
477 				forceClosing(e.getChannel());
478 			}
479 		} catch (HttpInvalidAuthenticationException e1) {
480 			if (handler != null) {
481 				status = handler.handleException(this, arguments, response, jsonObject, e1);
482 			}
483 			if (status == HttpResponseStatus.OK) {
484 				status = HttpResponseStatus.UNAUTHORIZED;
485 			}
486 			logger.warn("Error", e1);
487 			if (response.getDetail().isEmpty()) {
488 				response.setDetail(e1.getMessage());
489 			}
490 			if (handler != null) {
491 				finalizeSend(channel);
492 			} else {
493 				forceClosing(e.getChannel());
494 			}
495 		} catch (HttpNotFoundRequestException e1) {
496 			if (handler != null) {
497 				status = handler.handleException(this, arguments, response, jsonObject, e1);
498 			}
499 			if (status == HttpResponseStatus.OK) {
500 				status = HttpResponseStatus.NOT_FOUND;
501 			}
502 			logger.warn("Error", e1);
503 			if (response.getDetail().isEmpty()) {
504 				response.setDetail(e1.getMessage());
505 			}
506 			if (handler != null) {
507 				finalizeSend(channel);
508 			} else {
509 				forceClosing(e.getChannel());
510 			}
511 		}
512 	}

514 	/**
515 	 * Create the decoder
516 	 * @throws HttpIncorrectRequestException
517 	 */
518 	protected void createDecoder() throws HttpIncorrectRequestException {
519 		HttpMethod method = request.getMethod();
520         if (!method.equals(HttpMethod.HEAD)) {
521         	// in order decoder allows to parse
522             request.setMethod(HttpMethod.POST);
523         }
524 		try {
525 			decoder = new HttpPostRequestDecoder(factory, request);
526 		} catch (ErrorDataDecoderException e1) {
527 			status = HttpResponseStatus.NOT_ACCEPTABLE;
528 			throw new HttpIncorrectRequestException(e1);
529 		} catch (Exception e1) {
530 			// GETDOWNLOAD Method: should not try to create a HttpPostRequestDecoder
531 			// So OK but stop here
532 			status = HttpResponseStatus.NOT_ACCEPTABLE;
533 			throw new HttpIncorrectRequestException(e1);
534 		}
535 	}

537 	/**
538 	 * Read all InterfaceHttpData from finished transfer
539 	 * 
540 	 * @throws HttpIncorrectRequestException
541 	 */
542 	protected void readAllHttpData() throws HttpIncorrectRequestException {
543 		List<InterfaceHttpData> datas = null;
544 		try {
545 			datas = decoder.getBodyHttpDatas();
546 		} catch (NotEnoughDataDecoderException e1) {
547 			// Should not be!
548 			logger.warn("decoder issue", e1);
549 			status = HttpResponseStatus.NOT_ACCEPTABLE;
550 			throw new HttpIncorrectRequestException(e1);
551 		}
552 		for (InterfaceHttpData data : datas) {
553 			readHttpData(data);
554 		}
555 	}

557 	/**
558 	 * Read one Data
559 	 * 
560 	 * @param data
561 	 * @throws HttpIncorrectRequestException
562 	 */
563 	protected void readHttpData(InterfaceHttpData data)
564 			throws HttpIncorrectRequestException {
565 		if (data.getHttpDataType() == HttpDataType.Attribute) {
566 			ObjectNode body = arguments.getBody();
567 			try {
568 				body.put(data.getName(), ((Attribute) data).getValue());
569 			} catch (IOException e) {
570 				throw new HttpIncorrectRequestException("Bad reading", e);
571 			}
572 		} else if (data.getHttpDataType() == HttpDataType.FileUpload) {
573 			FileUpload fileUpload = (FileUpload) data;
574 			if (fileUpload.isCompleted()) {
575 				handler.getFileUpload(this, fileUpload, arguments, response);
576 			} else {
577 				logger.warn("File still pending but should not");
578 				fileUpload.delete();
579 				status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
580 				throw new HttpIncorrectRequestException("File still pending but should not");
581 			}
582 		} else {
583 			logger.warn("Unknown element: " + data.toString());
584 		}
585 	}

588 	/**
589 	 * To allow quick answer even if in very bad shape
590 	 * 
591 	 * @param channel
592 	 */
593 	protected void forceClosing(Channel channel) {
594 		if (status == HttpResponseStatus.OK) {
595 			status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
596 		}
597 		if (channel.isConnected()) {
598 			setWillClose(true);
599 			HttpResponse response = getResponse();
600 			response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "text/html");
601 			response.headers().set(HttpHeaders.Names.REFERER, request.getUri());
602 			String answer = "<html><body>Error " + status.getReasonPhrase() + "</body></html>";
603 			response.setContent(ChannelBuffers.wrappedBuffer(answer.getBytes(WaarpStringUtils.UTF8)));
604 			ChannelFuture future = channel.write(response);
605 			logger.debug("Will close");
606 			future.addListener(WaarpSslUtility.SSLCLOSE);
607 		}
608 		clean();
609 	}

612 	/**
613 	 * 
614 	 * @return the Http Response according to the status
615 	 */
616 	public HttpResponse getResponse() {
617 		// Decide whether to close the connection or not.
618 		if (request == null) {
619 			HttpResponse response = new DefaultHttpResponse(
620 					HttpVersion.HTTP_1_0, status);
621 			setCookies(response);
622 			setWillClose(true);
623 			return response;
624 		}
625 		boolean keepAlive = HttpHeaders.isKeepAlive(request);
626 		setWillClose(isWillClose() ||
627 				status != HttpResponseStatus.OK ||
628 				HttpHeaders.Values.CLOSE.equalsIgnoreCase(request
629 						.headers().get(HttpHeaders.Names.CONNECTION)) ||
630 				request.getProtocolVersion().equals(HttpVersion.HTTP_1_0) &&
631 				!keepAlive);
632 		if (isWillClose()) {
633 			keepAlive = false;
634 		}
635 		// Build the response object.
636 		HttpResponse response = new DefaultHttpResponse(
637 				request.getProtocolVersion(), status);
638 		if (keepAlive) {
639 			response.headers().set(HttpHeaders.Names.CONNECTION,
640 					HttpHeaders.Values.KEEP_ALIVE);
641 		}
642 		setCookies(response);
643 		return response;
644 	}

646 	/**
647 	 * Method that get a chunk of data
648 	 * 
649 	 * @param e
650 	 * @throws HttpIncorrectRequestException
651 	 * @throws HttpInvalidAuthenticationException 
652 	 * @throws HttpNotFoundRequestException 
653 	 */
654 	protected void bodyChunk(MessageEvent e) throws HttpIncorrectRequestException, HttpInvalidAuthenticationException, HttpNotFoundRequestException {
655 		// New chunk is received: only for Post!
656 		HttpChunk chunk = (HttpChunk) e.getMessage();
657 		if (handler.isBodyJsonDecoded()) {
658 			ChannelBuffer buffer = chunk.getContent();
659 			if (cumulativeBody != null) {
660 				cumulativeBody = ChannelBuffers.wrappedBuffer(cumulativeBody, buffer);
661 			} else {
662 				cumulativeBody = buffer;
663 			}
664 		} else {
665 			try {
666 				decoder.offer(chunk);
667 			} catch (ErrorDataDecoderException e1) {
668 				status = HttpResponseStatus.NOT_ACCEPTABLE;
669 				throw new HttpIncorrectRequestException(e1);
670 			}
671 			// example of reading chunk by chunk (minimize memory usage due to
672 			// Factory)
673 			readHttpDataChunkByChunk();
674 		}
675 		// example of reading only if at the end
676 		if (chunk.isLast()) {
677 			readingChunks = false;
678 			if (handler.isBodyJsonDecoded()) {
679 				jsonObject = getBodyJsonArgs(cumulativeBody);
680 				cumulativeBody = null;
681 			}
682 			response.setFromArgument(arguments);
683 			handler.endParsingRequest(this, arguments, response, jsonObject);
684 			finalizeSend(e.getChannel());
685 		}
