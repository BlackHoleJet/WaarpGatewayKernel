161     public static enum METHOD {
162     	/**
163     	 * REST: Standard GET item
164     	 * 
165     	 * The GET method means retrieve whatever information (in the form of an entity) 
166     	 * is identified by the Request-URI. If the Request-URI refers to a 
167     	 * data-producing process, it is the produced data which shall be returned as 
168     	 * the entity in the response and not the source text of the process, unless 
169     	 * that text happens to be the output of the process.
170     	 */
171     	GET(HttpMethod.GET),
172     	/**
173     	 * REST: Update existing item
174     	 * 
175     	 * The PUT method requests that the enclosed entity be stored under the 
176     	 * supplied Request-URI.
177     	 */
178     	PUT(HttpMethod.PUT),
179     	/**
180     	 * REST: Create a new item
181     	 * 
182     	 * The POST method is used to request that the origin server accept the 
183     	 * entity enclosed in the request as a new subordinate of the resource 
184     	 * identified by the Request-URI in the Request-Line.
185     	 */
186     	POST(HttpMethod.POST), 
187     	/**
188     	 * REST: Delete existing item
189     	 * 
190     	 * The DELETE method requests that the origin server delete the resource 
191     	 * identified by the Request-URI.
192     	 */
193     	DELETE(HttpMethod.DELETE), 
194     	/**
195     	 * REST: what options are supported for the URI
196     	 * 
197     	 * The OPTIONS method represents a request for information about 
198     	 * the communication options available on the request/response chain 
199     	 * identified by the Request-URI. This method allows the client to 
200     	 * determine the options and/or requirements associated with a resource, 
201     	 * or the capabilities of a server, without implying a resource action 
202     	 * or initiating a resource retrieval.
203     	 */
204     	OPTIONS(HttpMethod.OPTIONS), 
205     	/**
206     	 * REST: as GET but no BODY (existence ? metadata ?)
207     	 * 
208     	 * The HEAD method is identical to GET except that the server MUST NOT 
209     	 * return a message-body in the response.
210     	 */
211     	HEAD(HttpMethod.HEAD),
212     	/**
213     	 * REST: should not be used, use POST instead
214     	 * 
215     	 * The PATCH method requests that a set of changes described in the 
216     	 * request entity be applied to the resource identified by the Request-URI.
217     	 */
218     	PATCH(HttpMethod.PATCH),
219     	/**
220     	 * REST: unknown usage
221     	 * 
222     	 * The TRACE method is used to invoke a remote, application-layer 
223     	 * loop-back of the request message.
224     	 */
225     	TRACE(HttpMethod.TRACE),
226     	/**
227     	 * REST: unknown
228     	 * 
229     	 * This specification reserves the method name CONNECT for use with 
230     	 * a proxy that can dynamically switch to being a tunnel
231     	 */
232     	CONNECT(HttpMethod.CONNECT);
233     	
234     	public final HttpMethod method;
235     	private METHOD(HttpMethod method) {
236     		this.method = method;
237     	}

379 	 * To be used for instance to check correctness of connection<br>
380 	 * <br>
381 	 * for instance, if X-AUTH is included<br>
382 	 * check if in uri or header, X-AUTH is present and check X-AUTH argument (known, any key if present)<br>
383 	 * then get timestamp and check is correct (|curtime - timestamp| < maxinterval)<br>
384 	 * then get all uri args in alphabetic lower case order<br>
385 	 * hash using SHA-1 all args (in order including timestamp)<br>
386 	 * compare sha-1 hashedkey with the computed one
387 	 * 
388 	 * @param channel
389 	 * @throws HttpInvalidAuthenticationException
390 	 */

393     /**
394      * 
395 	 * for instance, if X-AUTH is included<br>
396 	 * check if in uri or header, X-AUTH is present and check X-AUTH argument (known, any key if present)<br>
397 	 * then get timestamp and check is correct (|curtime - timestamp| < maxinterval)<br>
398 	 * then get all uri args in alphabetic lower case order<br>
399 	 * hash using SHA-1 all args (in order including timestamp)<br>
400 	 * compare sha-1 hashedkey with the computed one
401 	 * @param arguments contains ARG_PATH, ARGS_URI
402 	 * @return the authentication string if any that should be compared with ARG_X_AUTH_KEY
403 	 * @throws HttpIncorrectRequestException
404 	 */
405 	public static String computeBaseAuthent(ObjectNode arguments, String extraKey) throws HttpIncorrectRequestException {
406 		TreeMap<String, String> treeMap = new TreeMap<String, String>();
407 		TextNode argpath = (TextNode) arguments.get(ARG_PATH);
408 		ObjectNode arguri = (ObjectNode) arguments.get(ARGS_URI);
409 		if (arguri == null) {
410 			throw new HttpIncorrectRequestException("Not enough argument");
411 		}
412 		Iterator<String> iteratorKey = arguri.fieldNames();
413 		while (iteratorKey.hasNext()) {
414 			String key = iteratorKey.next();
415 			if (key.equalsIgnoreCase(ARG_X_AUTH_KEY)) {
416 				continue;
417 			}
418 			String keylower = key.toLowerCase();
419 			ArrayNode values = (ArrayNode) arguri.get(key);
420 			for (JsonNode jsonNode : values) {
421 				treeMap.put(keylower, jsonNode.asText());
422 			}
423 		}
424 		Set<String> keys = treeMap.keySet();
425 		String concat = argpath.asText()+ (keys.isEmpty() ? "" : "?");
426 		boolean first = true;
427 		for (String keylower : keys) {
428 			if (first) {
429 				concat += keylower+"="+treeMap.get(keylower);
430 				first = false;
431 			} else {
432 				concat += "&"+keylower+"="+treeMap.get(keylower);
433 			}
434 		}
435 		if (extraKey != null) {
436 			concat += "&"+ARG_X_AUTH_INTERNALKEY+"="+extraKey;
437 		}
438 		// FIXME to encode using HMACSHA1 
439 		logger.debug("to sign: {}",concat);
440 		try {
441 			return hmacSha256.cryptToHex(concat);
442 		} catch (Exception e) {
443 			throw new HttpIncorrectRequestException(e);
444 		}
445 	}

447 	/**
448 	 * set values from URI into arguments.path(ARGS_URI)
449 	 * 
450 	 * @throws HttpIncorrectRequestException
451 	 */
452 	public static void getUriArgs(String uri, ObjectNode arguments) throws HttpIncorrectRequestException {
453 		QueryStringDecoder decoderQuery = new QueryStringDecoder(uri);
454 		String path = decoderQuery.getPath();
455 		arguments.put(ARG_PATH, path);
456 		// compute path main uri
457 		String basepath = path;
458 		int pos = basepath.indexOf('/');
459 		if (pos >= 0) {
460 			if (pos == 0) {
461 				int pos2 = basepath.indexOf('/', 1);
462 				if (pos2 < 0) {
463 					basepath = basepath.substring(1);
464 				} else {
465 					basepath = basepath.substring(1, pos2);
466 				}
467 			} else {
468 				basepath = basepath.substring(0, pos);
469 			}
470 		}
471 		arguments.put(ARG_BASEPATH, basepath);
472 		// compute sub path args
473 		if (pos == 0) {
474 			pos = path.indexOf('/', 1);
475 		}
476 		if (pos >= 0) {
477 			int pos2 = path.indexOf('/', pos+1);
478 			if (pos2 > 0) {
479 				ArrayNode array = arguments.putArray(ARGS_SUBPATH);
480 				while (pos2 > 0) {
481 					array.add(path.substring(pos+1, pos2));
482 					pos = pos2;
483 					pos2 = path.indexOf('/', pos+1);
484 				}
485 			}
486 		}
487 		Map<String, List<String>> map = decoderQuery.getParameters();
488 		ObjectNode node = arguments.putObject(ARGS_URI);
489 		for (String key : map.keySet()) {
490 			if (key.equalsIgnoreCase(ARG_X_AUTH_KEY)) {
491 				arguments.put(ARG_X_AUTH_KEY, map.get(key).get(0));
492 				continue;
493 			}
494 			if (key.equalsIgnoreCase(ARG_X_AUTH_USER)) {
495 				arguments.put(ARG_X_AUTH_USER, map.get(key).get(0));
496 				continue;
497 			}
498 			ArrayNode array = node.putArray(key);
499 			for (String val : map.get(key)) {
500 				array.add(val);
501 			}
502 		}
503 	}

507 	 * set values from Header into arguments.path(ARGS_HEADER)
508 	 * 
509 	 * @throws HttpIncorrectRequestException
510 	 */
511 	protected void getHeaderArgs() throws HttpIncorrectRequestException {
512 		ObjectNode node = arguments.putObject(ARGS_HEADER);
513 		List<Entry<String,String>> list = request.getHeaders();
514 		for (Entry<String, String> entry : list) {
515 			String key = entry.getKey();
516 			if (! key.equals(HttpHeaders.Names.COOKIE)) {
517 				if (key.equalsIgnoreCase(ARG_X_AUTH_KEY)) {
518 					arguments.put(ARG_X_AUTH_KEY, entry.getValue());
519 					continue;
520 				}
521 				if (key.equalsIgnoreCase(ARG_X_AUTH_USER)) {
522 					arguments.put(ARG_X_AUTH_USER, entry.getValue());
523 					continue;
524 				}
525 				node.put(entry.getKey(), entry.getValue());
526 			}
527 		}
528 	}

532 	 * set values from Cookies into arguments.path(ARGS_COOKIE)
533 	 * 
534 	 * @throws HttpIncorrectRequestException
535 	 */
536 	protected void getCookieArgs() throws HttpIncorrectRequestException {
537 		Set<Cookie> cookies;
538 		String value = request.getHeader(HttpHeaders.Names.COOKIE);
539 		if (value == null) {
540 			cookies = Collections.emptySet();
541 		} else {
542 			CookieDecoder decoder = new CookieDecoder();
543 			cookies = decoder.decode(value);
544 		}
545 		if (!cookies.isEmpty()) {
546 			ObjectNode node = arguments.putObject(ARGS_COOKIE);
547 			for (Cookie cookie : cookies) {
548 				node.put(cookie.getName(), cookie.getValue());
549 			}
550 		}
551 	}

553 	/**
554 	 * Method to set Cookies in httpResponse from response ObjectNode
555 	 * 
556 	 * @param httpResponse
557 	 */
558 	protected void setCookies(HttpResponse httpResponse) {
559 		if (response == null) {
560 			return;
561 		}
562 		JsonNode cookieON = response.path(ARGS_COOKIE);
563 		if (! cookieON.isMissingNode()) {
564 			CookieEncoder cookieEncoder = new CookieEncoder(true);
565 			Iterator<Entry<String, JsonNode>> iter = cookieON.fields();
566 			while (iter.hasNext()) {
567 				Entry<String, JsonNode> entry = iter.next();
568 				cookieEncoder.addCookie(entry.getKey(), entry.getValue().asText());
569 			}
570 			httpResponse.addHeader(HttpHeaders.Names.SET_COOKIE, cookieEncoder.encode());
571 		}
572 	}

625 	/**
626 	 * 
627 	 * @return RestMethodHandler associated with the current context
628 	 * @throws HttpIncorrectRequestException 
629 	 */
630 	protected RestMethodHandler getHandler() throws HttpIncorrectRequestException {
631 		METHOD method = getMethod();
632 		String uri = getBASEURI();
633 		if (DbConstant.admin != null) {
634 			WaarpActionLogger.logCreate(DbConstant.admin.session, "Request received: "
635 				+ arguments, session, uri+":"+method);
636 		}
637 		boolean restFound = false;
638 		RestMethodHandler handler = restHashMap.get(uri);
639 		if (handler != null) {
640 			response = JsonHandler.createObjectNode();
641 			handler.checkArgumentsCorrectness(this, getURI(), arguments, response);

700 			}

702 			// real error => 400

713 			if (handler != null) {
714 				ChannelFuture future = handler.sendResponse(this, channel, arguments, response, jsonObject, status);
715 				if (future != null) {

833 	 * 
834 	 * @return the Http Response according to the status
835 	 */
836 	public HttpResponse getResponse() {
837 		// Decide whether to close the connection or not.
838 		if (request == null) {
839 			HttpResponse response = new DefaultHttpResponse(
840 					HttpVersion.HTTP_1_0, status);
841 			setCookies(response);
842 			setWillClose(true);
843 			return response;
844 		}
845 		boolean keepAlive = HttpHeaders.isKeepAlive(request);
846 		setWillClose(isWillClose() ||
847 				status != HttpResponseStatus.OK ||
848 				HttpHeaders.Values.CLOSE.equalsIgnoreCase(request
849 						.getHeader(HttpHeaders.Names.CONNECTION)) ||
850 				request.getProtocolVersion().equals(HttpVersion.HTTP_1_0) &&
851 				!keepAlive);
852 		if (isWillClose()) {
853 			keepAlive = false;
854 		}
855 		// Build the response object.
856 		HttpResponse response = new DefaultHttpResponse(
857 				request.getProtocolVersion(), status);
858 		if (keepAlive) {
859 			response.setHeader(HttpHeaders.Names.CONNECTION,
860 					HttpHeaders.Values.KEEP_ALIVE);
861 		}
862 		setCookies(response);
863 		return response;
864 	}

868 	 * 
869 	 * @param e

872 	protected void bodyChunk(MessageEvent e) throws HttpIncorrectRequestException {
873 		// New chunk is received: only for Post!
874 		HttpChunk chunk = (HttpChunk) e.getMessage();
875 		if (handler.isBodyDedicatedDecode()) {
876 			ChannelBuffer buffer = chunk.getContent();
877 			logger.warn("new chunk");
878 			if (cumulativeBody != null) {
879 				cumulativeBody = ChannelBuffers.wrappedBuffer(cumulativeBody, buffer);
880 			} else {
881 				cumulativeBody = buffer;
882 			}
883 		} else {
884 			try {
885 				decoder.offer(chunk);
886 			} catch (ErrorDataDecoderException e1) {
887 				status = HttpResponseStatus.NOT_ACCEPTABLE;
888 				throw new HttpIncorrectRequestException(e1);
889 			}
890 			// example of reading chunk by chunk (minimize memory usage due to
891 			// Factory)
892 			readHttpDataChunkByChunk();
893 		}
894 		// example of reading only if at the end
895 		if (chunk.isLast()) {
896 			readingChunks = false;
897 			jsonObject = getBodyJsonArgs(cumulativeBody);
898 			cumulativeBody = null;
899 			handler.endBody(this, arguments, response, jsonObject);
900 			ChannelFuture future = handler.sendResponse(this, e.getChannel(), arguments, response, jsonObject, status);
901 			if (future != null) {

903 			}
