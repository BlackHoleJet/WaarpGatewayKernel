50  	/**
51  	 * Write a file, taking into account cache enabled and removing session cookie
52  	 * 
53  	 * @deprecated should not be used. Use writeFileChunked instead
54  	 * @param request
55  	 * @param channel
56  	 * @param filename
57  	 * @param cookieNameToRemove
58  	 */
59  	protected static void writeFile(HttpRequest request, Channel channel,
60  			String filename, String cookieNameToRemove) {
61  		// Convert the response content to a ChannelBuffer.
62  		HttpResponse response;
63  		if (request.containsHeader(HttpHeaders.Names.IF_MODIFIED_SINCE)) {
64  			response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
65  					HttpResponseStatus.NOT_MODIFIED);
66  			handleCookies(request, response, cookieNameToRemove);
67  			channel.write(response);
68  			return;
69  		}
70  		File file = new File(filename);
71  		byte[] bytes = new byte[(int) file.length()];
72  		FileInputStream fileInputStream;
73  		try {
74  			fileInputStream = new FileInputStream(file);
75  		} catch (FileNotFoundException e) {
76  			response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
77  					HttpResponseStatus.NOT_FOUND);
78  			handleCookies(request, response, cookieNameToRemove);
79  			channel.write(response);
80  			return;
81  		}
82  		try {
83  			fileInputStream.read(bytes);
84  		} catch (IOException e) {
85  			try {
86  				fileInputStream.close();
87  			} catch (IOException e1) {
88  			}
89  			response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
90  					HttpResponseStatus.NOT_FOUND);
91  			handleCookies(request, response, cookieNameToRemove);
92  			channel.write(response);
93  			return;
94  		}
95  		try {
96  			fileInputStream.close();
97  		} catch (IOException e) {
98  		}
99  		ChannelBuffer buf = ChannelBuffers.copiedBuffer(bytes);
100 		// Build the response object.
101 		response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
102 				HttpResponseStatus.OK);
103 		response.setContent(buf);
104 		response.setHeader(HttpHeaders.Names.CONTENT_LENGTH,
105 				String.valueOf(buf.readableBytes()));
106 		ArrayList<String> cache_control = new ArrayList<String>(2);
107 		cache_control.add(HttpHeaders.Values.PUBLIC);
108 		cache_control.add(HttpHeaders.Values.MAX_AGE + "=" + 31536000);// 1 year
109 		response.setHeader(HttpHeaders.Names.CACHE_CONTROL, cache_control);
110 		response.setHeader(HttpHeaders.Names.LAST_MODIFIED,
111 				"Tue, 15 Nov 1994 12:45:26 GMT");
112 		handleCookies(request, response, cookieNameToRemove);
113 		// Write the response.
114 		channel.write(response);
115 	}

117 	/**
118 	 * Write a file, taking into account cache enabled and removing session cookie
119 	 * 
120 	 * @param request
121 	 * @param channel
122 	 * @param filename
123 	 * @param cookieNameToRemove
124 	 */
125 	public static void writeFileChunked(HttpRequest request, Channel channel,
126 			String filename, String cookieNameToRemove) {
127 		// Convert the response content to a ChannelBuffer.
128 		HttpResponse response;
129 		if (request.containsHeader(HttpHeaders.Names.IF_MODIFIED_SINCE)) {
130 			response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
131 					HttpResponseStatus.NOT_MODIFIED);
132 			handleCookies(request, response, cookieNameToRemove);
133 			channel.write(response);
134 			return;
135 		}
136 		File file = new File(filename);
137 		if (!file.isFile() || !file.canRead()) {
138 			response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
139 					HttpResponseStatus.NOT_FOUND);
140 			handleCookies(request, response, cookieNameToRemove);
141 			channel.write(response);
142 			return;
143 		}
144 		long size = file.length();
145 		ChunkedNioFile nioFile;
146 		try {
147 			nioFile = new ChunkedNioFile(file);
148 		} catch (IOException e) {
149 			response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
150 					HttpResponseStatus.NOT_FOUND);
151 			handleCookies(request, response, cookieNameToRemove);
152 			channel.write(response);
153 			return;
154 		}
155 		response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
156 				HttpResponseStatus.OK);
157 		response.setHeader(HttpHeaders.Names.CONTENT_LENGTH,
158 				String.valueOf(size));
159 		ArrayList<String> cache_control = new ArrayList<String>(2);
160 		cache_control.add(HttpHeaders.Values.PUBLIC);
161 		cache_control.add(HttpHeaders.Values.MAX_AGE + "=" + 31536000);// 1 year
162 		response.setHeader(HttpHeaders.Names.CACHE_CONTROL, cache_control);
163 		response.setHeader(HttpHeaders.Names.LAST_MODIFIED,
164 				"Tue, 15 Nov 1994 12:45:26 GMT");
165 		handleCookies(request, response, cookieNameToRemove);
166 		// Write the response.
167 		channel.write(response);
168 		channel.write(nioFile);
169 	}

171 	/**
172 	 * Remove the given named cookie
173 	 * 
174 	 * @param request
175 	 * @param response
176 	 * @param cookieNameToRemove
177 	 */
178 	public static void handleCookies(HttpRequest request,
179 			HttpResponse response, String cookieNameToRemove) {
180 		String cookieString = request.getHeader(HttpHeaders.Names.COOKIE);
181 		if (cookieString != null) {
182 			CookieDecoder cookieDecoder = new CookieDecoder();
183 			Set<Cookie> cookies = cookieDecoder.decode(cookieString);
184 			if (!cookies.isEmpty()) {
185 				// Reset the sessions if necessary.
186 				CookieEncoder cookieEncoder = new CookieEncoder(true);
187 				// Remove all Session for images
188 				for (Cookie cookie : cookies) {
189 					if (cookie.getName().equalsIgnoreCase(cookieNameToRemove)) {
190 					} else {
191 						cookieEncoder.addCookie(cookie);
192 						response.addHeader(HttpHeaders.Names.SET_COOKIE, cookieEncoder.encode());
193 						cookieEncoder = new CookieEncoder(true);
194 					}
195 				}
196 			}
197 		}
198 	}
