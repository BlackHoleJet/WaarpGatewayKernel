27  /**
28   * @author "Frederic Bregier"
29   * 
30   */
31  public class HttpPageHandler {
32  	/*
33  	 * Need as default error pages: 400, 401, 403, 404, 406, 500
34  	 */
35  
36  	public static String hostid;
37  
38  	public HashMap<String, HttpPage> hashmap;
39  
40  	/**
41  	 * @param hashmap
42  	 */
43  	public HttpPageHandler(HashMap<String, HttpPage> hashmap) {
44  		this.hashmap = hashmap;
45  	}
46  
47  	/**
48  	 * 
49  	 * @param code
50  	 * @return an HttpPage according to the error code (400, 404, 500, ...)
51  	 */
52  	public HttpPage getHttpPage(int code) {
53  		String scode = Integer.toString(code);
54  		return this.hashmap.get(scode);
55  	}
56  
57  	/**
58  	 * 
59  	 * @param uri
60  	 * @param method
61  	 * @param session
62  	 * @return the associated HttpPage if any
63  	 * @throws HttpIncorrectRequestException
64  	 */
65  	public HttpPage getHttpPage(String uri, String method, HttpSession session)
66  			throws HttpIncorrectRequestException {
67  		HttpPage page = this.hashmap.get(uri);
68  		if (page == null) {
69  			return null;
70  		}
71  		switch (page.pagerole) {
72  			case DELETE:
73  				if (!method.equalsIgnoreCase("DELETE")) {
74  					// error
75  					WaarpActionLogger.logErrorAction(DbConstant.admin.session, session,
76  							"Incorrect Page: " + page.pagerole, HttpResponseStatus.BAD_REQUEST);
77  					if (page.errorpage != null || page.errorpage.length() > 1) {
78  						page = this.hashmap.get(page.errorpage);
79  					} else {
80  						page = null;
81  					}
82  				}
83  				break;
84  			case HTML:
85  			case MENU:
86  				// no check
87  				break;
88  			case GETDOWNLOAD:
89  				if (!method.equalsIgnoreCase("GET")) {
90  					// error
91  					WaarpActionLogger.logErrorAction(DbConstant.admin.session, session,
92  							"Incorrect Page: " + page.pagerole, HttpResponseStatus.BAD_REQUEST);
93  					if (page.errorpage != null || page.errorpage.length() > 1) {
94  						page = this.hashmap.get(page.errorpage);
95  					} else {
96  						page = null;
97  					}
98  				}
99  				break;
100 			case POST:
101 			case POSTUPLOAD:
102 				if (!method.equalsIgnoreCase("POST")) {
103 					// error
104 					WaarpActionLogger.logErrorAction(DbConstant.admin.session, session,
105 							"Incorrect Page: " + page.pagerole, HttpResponseStatus.BAD_REQUEST);
106 					if (page.errorpage != null || page.errorpage.length() > 1) {
107 						page = this.hashmap.get(page.errorpage);
108 					} else {
109 						page = null;
110 					}
111 				}
112 				break;
113 			case PUT:
114 				if (!method.equalsIgnoreCase("PUT")) {
115 					// error
116 					WaarpActionLogger.logErrorAction(DbConstant.admin.session, session,
117 							"Incorrect Page: " + page.pagerole, HttpResponseStatus.BAD_REQUEST);
118 					if (page.errorpage != null || page.errorpage.length() > 1) {
119 						page = this.hashmap.get(page.errorpage);
120 					} else {
121 						page = null;
122 					}
123 				}
124 				break;
125 			case ERROR:
126 				break;
127 			default:
128 				// error
129 				WaarpActionLogger.logErrorAction(DbConstant.admin.session, session,
130 						"Incorrect Page: " + page.pagerole, HttpResponseStatus.BAD_REQUEST);
131 				if (page.errorpage != null || page.errorpage.length() > 1) {
132 					page = this.hashmap.get(page.errorpage);
133 				} else {
134 					page = null;
135 				}
136 		}
137 		if (page == null) {
138 			throw new HttpIncorrectRequestException("No Page found");
139 		}
140 		return page;
141 	}
